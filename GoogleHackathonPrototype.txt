  const canvas = document.getElementById('bg-canvas');
  const ghostCanvas = document.getElementById('ghost-bg-canvas');
  const ctx = canvas.getContext('2d');
  const gCtx = ghostCanvas.getContext('2d');
  
  let particles = [];
  function initParticles() {
    canvas.width = window.innerWidth; canvas.height = window.innerHeight;
    ghostCanvas.width = window.innerWidth; ghostCanvas.height = window.innerHeight;
    particles = [];
    // Kept around 60 particles for dense, bright connections
    for (let i = 0; i < 60; i++) {
      particles.push({ 
        x: Math.random()*canvas.width, 
        y: Math.random()*canvas.height, 
        vx: (Math.random()-0.5)*0.3, 
        vy: (Math.random()-0.5)*0.3,
        phase: Math.random() * Math.PI * 2 
      });
    }
  }

  function drawParticles() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    gCtx.clearRect(0, 0, ghostCanvas.width, ghostCanvas.height);
    
    // BRIGHTER SETTINGS: Increased alpha values from 0.2/0.04 to 0.5/0.15
    ctx.fillStyle = "rgba(34, 211, 238, 0.5)"; 
    ctx.strokeStyle = "rgba(34, 211, 238, 0.15)";
    gCtx.fillStyle = "rgba(34, 211, 238, 0.5)"; 
    gCtx.strokeStyle = "rgba(34, 211, 238, 0.2)";

    let time = Date.now() * 0.001;

    particles.forEach(p => {
      let fx = Math.sin(time + p.phase) * 0.1;
      let fy = Math.cos(time + p.phase) * 0.1;
      
      p.x += p.vx + fx; 
      p.y += p.vy + fy;

      if (p.x < 0 || p.x > canvas.width) p.vx *= -1;
      if (p.y < 0 || p.y > canvas.height) p.vy *= -1;
      
      // Slightly larger dots for brightness
      ctx.beginPath(); ctx.arc(p.x, p.y, 1.2, 0, Math.PI*2); ctx.fill();
      gCtx.beginPath(); gCtx.arc(p.x, p.y, 1.5, 0, Math.PI*2); gCtx.fill();

      particles.forEach(p2 => {
        let dist = Math.hypot(p.x - p2.x, p.y - p2.y);
        // Slightly longer connection range for a "fuller" look
        if (dist < 200) {

  function spawnAttack() {
    const from = hubKeys[Math.floor(Math.random() * hubKeys.length)];
    const to = hubKeys[Math.floor(Math.random() * hubKeys.length)];
    if(from === to) return;
    const start = hubs[from]; const end = hubs[to];
    const mid = [ (start[0] + end[0]) / 2 + (Math.random() - 0.5) * 30, (start[1] + end[1]) / 2 + (Math.random() - 0.5) * 30 ];
    const path = L.curve(['M', start, 'Q', mid, end], { color: '#22d3ee', weight: 2, fill: false, className: 'attack-path' }).addTo(map);
    setTimeout(() => {
      L.marker(end, { icon: threatIcon }).addTo(map);
      addWarLog(`THREAT_DETECTED: Source ${from} -> Target ${to}`, true);
